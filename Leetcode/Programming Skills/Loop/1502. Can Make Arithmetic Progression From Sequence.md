```python3
class Solution:
    def canMakeArithmeticProgression(self, arr: List[int]) -> bool:
        arr.sort()
        a= abs(arr[0]-arr[1]) 
        l= len(arr)
        for i in range(1,l-1):
            if arr[i+1]-arr[i]==a :
                continue
            else :
                return False
        
        return True
```

```python3
class Solution:
    def canMakeArithmeticProgression(self, arr: List[int]) -> bool:
        arr.sort()
        # 반복문으로 돌려서 결과를 set 자료구조로 
        # 그러므로 set은 반복을 허용안하니
        # 결국 하나의 element 만 있으면  
        return len(set(arr[i-1] - arr[i] for i in range(1, len(arr)))) == 1
```
